    class Rectangle:
        def __init__(length,width,name):
            self.length=length
            self.width=width
            self.name=name
        def get_area(self):
            return self.length*self.width
        def compare_size(self,other):
            if self.get_area<other.get_area:
                print(other.name,"is larger")
            else:
                print(self.name,"is larger")
